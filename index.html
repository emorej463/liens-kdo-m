  <script>

    function toggleDispo(button) {

      const lien = button.previousElementSibling;

      lien.classList.toggle("indisponible");


      if (lien.classList.contains("indisponible")) {

        button.textContent = "Rendre dispo";

      } else {

        button.textContent = "Plus dispo";

      }

    }

  </script>
